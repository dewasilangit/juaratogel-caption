  <script>
    const generateBtn = document.getElementById('generateBtn');
    const inputText = document.getElementById('inputText');
    const loading = document.getElementById('loading');
    const result = document.getElementById('result');
    const captionOutput = document.getElementById('captionOutput');
    const copyBtn = document.getElementById('copyBtn');

    // Daftar template caption (bisa kamu tambah sendiri)
    const templates = [
      "🔥 {keyword} bikin hari ini makin seru! Siapa yang ikutan? 👀",
      "Gak nyangka {keyword} bisa segini enaknya! ✅ Recommended banget!",
      "Pagi-pagi udah excited sama {keyword}. Siap gaspol hari ini! 💪",
      "Baru coba {keyword} dan langsung ketagihan! Kamu udah pernah? 😍",
      "Tips juara: {keyword} adalah kunci sukses di 2026! 📈",
      "Mau viral? Coba posting tentang {keyword} dengan gaya ini 🔥",
      "Dari nol sampai hero berkat {keyword}. Cerita lengkap di thread ini 👇",
      "Siapa bilang {keyword} susah? Ternyata gampang banget kok! 😉"
    ];

    function generateCaption(keyword) {
      if (!keyword.trim()) {
        return "Masukkan kata kunci dulu ya biar aku bisa bantu buat caption keren! ✨";
      }

      const randomTemplate = templates[Math.floor(Math.random() * templates.length)];
      let caption = randomTemplate.replace("{keyword}", keyword);

      // Tambah variasi emoji dan hashtag
      const hashtags = ["#CaptionJuara", "#TwitterTips", "#ContentCreator", "#Viral"];
      caption += "\n\n" + hashtags.join(" ");

      return caption;
    }

    generateBtn.addEventListener('click', () => {
      const keyword = inputText.value.trim();

      // Tampilkan loading
      loading.style.display = 'block';
      result.classList.remove('show');

      // Simulasi delay generate (seperti AI)
      setTimeout(() => {
        const generated = generateCaption(keyword);
        captionOutput.textContent = generated;
        
        loading.style.display = 'none';
        result.classList.add('show');
      }, 1200);
    });

    // Copy ke clipboard
    copyBtn.addEventListener('click', () => {
      const text = captionOutput.textContent;
      navigator.clipboard.writeText(text).then(() => {
        const originalText = copyBtn.textContent;
        copyBtn.textContent = "✅ Berhasil dicopy!";
        setTimeout(() => {
          copyBtn.textContent = originalText;
        }, 2000);
      });
    });

    // Enter key support
    inputText.addEventListener('keypress', (e) => {
      if (e.key === 'Enter' && !e.shiftKey) {
        e.preventDefault();
        generateBtn.click();
      }
    });
  </script>
